        statusEl.classList.add('hidden');

        try {
            // Validasi file
            const fotoFile = document.getElementById('foto_peserta').files[0];
            const buktiFile = document.getElementById('bukti_pembayaran').files[0];
            
            if (!fotoFile || !buktiFile) {
                throw new Error('Upload foto dan bukti pembayaran wajib.');
            }
            
            if (fotoFile.size > 2 * 1024 * 1024) {
                throw new Error('Foto maksimal 2MB.');
            }
            
            if (buktiFile.size > 5 * 1024 * 1024) {
                throw new Error('Bukti pembayaran maksimal 5MB.');
            }
            
            const allowedFoto = ['image/jpeg', 'image/png'];
            const allowedBukti = ['image/jpeg', 'image/png', 'application/pdf'];
            
            if (!allowedFoto.includes(fotoFile.type)) {
                throw new Error('Format foto harus JPG/PNG.');
            }
            
            if (!allowedBukti.includes(buktiFile.type)) {
                throw new Error('Format bukti harus JPG/PNG/PDF.');
            }

            // Baca file ke base64
            const fotoBase64 = await readFileAsBase64(fotoFile);
            const buktiBase64 = await readFileAsBase64(buktiFile);

            // Kumpulkan data
            const payload = {
                nama: document.getElementById('nama').value.trim(),
                jenis_kelamin: document.getElementById('jenis_kelamin').value,
                tempat_lahir: document.getElementById('tempat_lahir').value.trim(),
                tanggal_lahir: document.getElementById('tanggal_lahir').value,
                sekolah: document.getElementById('sekolah').value.trim(),
                pembimbing: document.getElementById('pembimbing').value.trim(),
                wa_peserta: document.getElementById('wa_peserta').value.trim(),
                wa_orangtua: document.getElementById('wa_orangtua').value.trim(),
                alamat: document.getElementById('alamat').value.trim(),
                kategori: document.getElementById('kategori').value,
                surah_awal: document.getElementById('surah_awal').value.trim(),
                surah_akhir: document.getElementById('surah_akhir').value.trim(),
                foto: fotoBase64,
                foto_nama: fotoFile.name,
                foto_tipe: fotoFile.type,
                bukti: buktiBase64,
                bukti_nama: buktiFile.name,
                bukti_tipe: buktiFile.type,
            };

            // Kirim ke API
            const response = await fetch(API_URL + '?action=register', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload),
            });
            
            const result = await response.json();

            if (result.success) {
                Swal.fire({
                    icon: 'success',
                    title: 'Pendaftaran Berhasil!',
                    html: `<strong>Nomor Registrasi:</strong> ${result.data.registration_id || 'THF-2026-xxxx'}<br/><span class="text-sm text-gray-500">Status: Menunggu Verifikasi</span>`,
                    confirmButtonColor: '#2563eb',
                    confirmButtonText: 'OK',
                });
                form.reset();
                statusEl.innerHTML = `<span class="text-green-600"><i class="fas fa-check-circle"></i> Pendaftaran berhasil! ID: ${result.data.registration_id || 'THF-2026-xxxx'}</span>`;
                statusEl.classList.remove('hidden');
            } else {
                throw new Error(result.message || 'Gagal mendaftar.');
            }
        } catch (err) {
            Swal.fire({
                icon: 'error',
                title: 'Oops!',
                text: err.message,
                confirmButtonColor: '#2563eb'
            });
            statusEl.innerHTML = `<span class="text-red-500"><i class="fas fa-exclamation-circle"></i> ${err.message}</span>`;
            statusEl.classList.remove('hidden');
        } finally {
            btn.disabled = false;
            btn.innerHTML = '<i class="fas fa-paper-plane"></i> Daftar Sekarang';
        }
    });

    function readFileAsBase64(file) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.onload = () => resolve(reader.result.split(',')[1]);
            reader.onerror = reject;
            reader.readAsDataURL(file);
        });
    }
